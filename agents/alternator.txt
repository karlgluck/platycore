--- SCRIPT ---
Set up a SCRIPT field that either targets the note ONE or TWO. Each one swaps control to the next for the next iteration.

   SELECT "C1:D1"
    FIELD "SCRIPT"
     TEXT "ONE"
   SELECT "E1:H1"
     TEXT "SCRIPT"
   SELECT "C1:H1"
    PANEL "5"

   SELECT "C2:E2"
     TEXT "ONE"
    PANEL "5"
     NOTE "ONE"
        + "agent.WriteField('SCRIPT', 'TWO');"

   SELECT "F2:H2"
     TEXT "TWO"
    PANEL "5"
     NOTE "TWO"
        + "agent.WriteField('SCRIPT', 'ONE');"
